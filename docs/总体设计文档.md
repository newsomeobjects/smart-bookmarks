## 3. 技术选型

### 3.1 技术栈

| 类别 | 技术选型 | 说明 | 实现状态 |
|------|----------|------|----------|
| 扩展框架 | Chrome Extension Manifest V3 | 最新扩展标准 | ✅ |
| 开发语言 | JavaScript (ES6+) | 原生开发,无框架依赖 | ✅ |
| UI框架 | 原生HTML/CSS | 轻量化 | ✅ |
| 数据存储 | chrome.storage.local | 浏览器本地存储 | ✅ |
| 书签API | chrome.bookmarks | 原生书签操作API | ✅ |
| 标签页API | chrome.tabs | 访问追踪API | ✅ |
| AI接口 | RESTful API | 支持多种AI模型 | ✅ |
| 提示词管理 | JSON模板 | 灵活的提示词配置 | ✅ |

### 3.2 AI模型支持

| 提供商 | 端点 | 模型示例 | 兼容性 |
|--------|------|----------|--------|
| DeepSeek | https://api.deepseek.com/v1/chat/completions | deepseek-chat | ✅ 已测试 |
| OpenAI | https://api.openai.com/v1/chat/completions | gpt-4, gpt-3.5-turbo | ✅ 兼容 |
| Anthropic | https://api.anthropic.com/v1/messages | claude-3-opus | ⚠️ 需适配 |
| 自定义 | 用户配置 | 任意兼容OpenAI格式的API | ✅ 支持 |

### 3.3 关键技术决策

| 决策点 | 选择 | 理由 |
|--------|------|------|
| 分类策略 | 两阶段分类 | 先生成分类树,再分配书签,提高分类质量 |
| 缓存策略 | 本地缓存+定期刷新 | 减少API调用,提升响应速度 |
| 分批处理 | 30个书签/批次 | 平衡API调用次数和响应时间 |
| 延迟处理 | 5秒延迟 | 避免用户正在编辑书签时触发分类 |
| 提示词管理 | JSON模板 | 灵活配置,支持多语言和优化 |

### 6.3 分类结果校验算法 ✅ 已实现

```javascript
function validateClassificationResult(result, bookmarks, validPaths) {
    const errors = [];
    
    // 1. 检查完整性
    if (!result.assignments || !Array.isArray(result.assignments)) {
        errors.push('缺少assignments数组');
        return { valid: false, errors };
    }
    
    // 2. 检查数量匹配
    if (result.assignments.length !== bookmarks.length) {
        errors.push(`归类数量不匹配: 期望${bookmarks.length}个,实际${result.assignments.length}个`);
    }
    
    // 3. 检查每个分类
    const bookmarkIds = new Set(bookmarks.map(b => b.id));
    const assignedIds = new Set();
    
    for (const assignment of result.assignments) {
        // 检查ID有效性
        if (!bookmarkIds.has(assignment.id)) {
            errors.push(`无效的书签ID: ${assignment.id}`);
        }
        assignedIds.add(assignment.id);
        
        // 检查路径有效性
        if (assignment.path !== '未分类' && !validPaths.has(assignment.path)) {
            errors.push(`无效的分类路径: ${assignment.path}`);
        }
        
        // 检查理由完整性
        if (!assignment.reason || assignment.reason.trim().length === 0) {
            errors.push(`书签${assignment.id}缺少归类理由`);
        }
    }
    
    // 4. 检查遗漏
    const missingIds = [...bookmarkIds].filter(id => !assignedIds.has(id));
    if (missingIds.length > 0) {
        errors.push(`未归类的书签ID: ${missingIds.join(', ')}`);
    }
    
    return {
        valid: errors.length === 0,
        errors,
        stats: {
            total: bookmarks.length,
            classified: assignedIds.size,
            withReasons: result.assignments.filter(a => a.reason).length
        }
    };
}
```

### 6.4 热度计算算法 🔮 V2.0规划

```javascript
function calculateHeatScore(visitRecord, config) {
    const now = Date.now();
    const daysSinceFirstVisit = (now - visitRecord.firstVisit) / (24 * 60 * 60 * 1000);
    const daysSinceLastVisit = (now - visitRecord.lastVisit) / (24 * 60 * 60 * 1000);
    
    // 计算访问频率分数
    const frequencyScore = visitRecord.visitCount / Math.max(daysSinceFirstVisit, 1);
    
    // 计算最近访问分数(指数衰减)
    const recencyScore = Math.exp(-daysSinceLastVisit / 7);
    
    // 综合热度分数
    const heatScore = 
        frequencyScore * config.visitWeight + 
        recencyScore * config.recencyWeight;
    
    // 应用时间衰减
    return heatScore * Math.pow(config.decayFactor, daysSinceLastVisit);
}
```

### 6.5 URL匹配算法 🔮 V2.0规划

```javascript
function matchUrlWithBookmark(visitUrl, bookmarkUrl) {
    try {
        const visitUrlObj = new URL(visitUrl);
        const bookmarkUrlObj = new URL(bookmarkUrl);
        
        // 完全匹配
        if (visitUrl === bookmarkUrl) {
            return { matched: true, score: 1.0 };
        }
        
        // 域名+路径匹配(忽略查询参数和hash)
        if (visitUrlObj.origin === bookmarkUrlObj.origin &&
            visitUrlObj.pathname === bookmarkUrlObj.pathname) {
            return { matched: true, score: 0.9 };
        }
        
        // 仅域名匹配
        if (visitUrlObj.hostname === bookmarkUrlObj.hostname) {
            return { matched: true, score: 0.5 };
        }
        
        return { matched: false, score: 0 };
    } catch (error) {
        return { matched: false, score: 0 };
    }
}
```
